```python
from web_agent import Agent, Browser, Chatwebagent

browser = Browser(
	headless=False,  # Show browser window
	window_size={'width': 1000, 'height': 700},  # Set window size
)

agent = Agent(
	task='Search for web agent',
	browser=browser,
	llm=Chatwebagent(),
)


async def main():
	await agent.run()
```
